16/// Return if current platform support virtualization extension.
17pub fn has_hardware_support() -> bool {
18    if let Some(feature) = raw_cpuid::CpuId::new().get_feature_info() {
19        feature.has_vmx()
20    } else {
21        false
22    }
23}

29fn as_axerr(err: x86::vmx::VmFail) -> axerrno::AxError {
30    use x86::vmx::VmFail;
31    match err {
32        VmFail::VmFailValid => ax_err_type!(BadState, vmcs::instruction_error().as_str()),
33        VmFail::VmFailInvalid => ax_err_type!(BadState, "VMCS pointer is not valid"),
34    }

19macro_rules! vmcs_read {
20    ($field_enum: ident, u64) => {
21        impl $field_enum {
22            pub fn read(self) -> AxResult<u64> {
23                #[cfg(target_pointer_width = "64")]
24                unsafe {
25                    vmx::vmread(self as u32).map_err(as_axerr)
26                }
27                #[cfg(target_pointer_width = "32")]
28                unsafe {
29                    let field = self as u32;
30                    Ok(vmx::vmread(field).map_err(as_axerr)?
31                        + (vmx::vmread(field + 1).map_err(as_axerr)? << 32))
32                }
33            }
34        }
35    };
36    ($field_enum: ident, $ux: ty) => {
37        impl $field_enum {
38            pub fn read(self) -> AxResult<$ux> {
39                unsafe { vmx::vmread(self as u32).map(|v| v as $ux).map_err(as_axerr) }
40            }
41        }
42    };
43}

45macro_rules! vmcs_write {
46    ($field_enum: ident, u64) => {
47        impl $field_enum {
48            pub fn write(self, value: u64) -> AxResult {
49                #[cfg(target_pointer_width = "64")]
50                unsafe {
51                    vmx::vmwrite(self as u32, value).map_err(as_axerr)
52                }
53                #[cfg(target_pointer_width = "32")]
54                unsafe {
55                    let field = self as u32;
56                    vmx::vmwrite(field, value & 0xffff_ffff).map_err(as_axerr)?;
57                    vmx::vmwrite(field + 1, value >> 32).map_err(as_axerr)?;
58                    Ok(())
59                }
60            }
61        }
62    };
63    ($field_enum: ident, $ux: ty) => {
64        impl $field_enum {
65            pub fn write(self, value: $ux) -> AxResult {
66                unsafe { vmx::vmwrite(self as u32, value as u64).map_err(as_axerr) }
67            }
68        }
69    };
70}

502/// VM-Entry/VM-Exit Interruption-Information Field. (SDM Vol. 3C, Section 24.8.3, 24.9.2)
503#[derive(Debug)]
504pub struct VmxInterruptInfo {
505    /// Vector of interrupt or exception.
506    pub vector: u8,
507    /// Determines details of how the injection is performed.
508    pub int_type: VmxInterruptionType,
509    /// For hardware exceptions that would have delivered an error code on the stack.
510    pub err_code: Option<u32>,
511    /// Whether the field is valid.
512    pub valid: bool,
513}
514
515impl VmxInterruptInfo {
516    /// Convert from the interrupt vector and the error code.
517    pub fn from(vector: u8, err_code: Option<u32>) -> Self {
518        Self {
519            vector,
520            int_type: VmxInterruptionType::from_vector(vector),
521            err_code,
522            valid: true,
523        }
524    }
525
526    /// Raw bits for writing to VMCS.
527    pub fn bits(&self) -> u32 {
528        let mut bits = self.vector as u32;
529        bits |= (self.int_type as u32) << 8;
530        bits.set_bit(11, self.err_code.is_some());
531        bits.set_bit(31, self.valid);
532        bits
533    }
534}

604
605impl TryFrom<u8> for ApicAccessExitType {
606    type Error = ();
607
608    fn try_from(value: u8) -> Result<Self, Self::Error> {
609        match value {
610            0 => Ok(Self::LinearDataRead),
611            1 => Ok(Self::LinearDataWrite),
612            2 => Ok(Self::LinearInstructionFetch),
613            3 => Ok(Self::LinearEventDelivery),
614            4 => Ok(Self::LinearMonitoring),
615            10 => Ok(Self::GuestPhysicalEventDelivery),
616            11 => Ok(Self::GuestPhysicalMonitoring),
617            15 => Ok(Self::GuestPhysicalInstructionFetchReadWrite),
618            _ => Err(()),
619        }
620    }
621}

639pub fn set_control(
640    control: VmcsControl32,
641    capability_msr: Msr,
642    old_value: u32,
643    set: u32,
644    clear: u32,
645) -> AxResult {
646    let cap = capability_msr.read();
647    let allowed0 = cap as u32;
648    let allowed1 = (cap >> 32) as u32;
649    assert_eq!(allowed0 & allowed1, allowed0);
650    debug!(
651        "set {:?}: {:#x} (+{:#x}, -{:#x})",
652        control, old_value, set, clear
653    );
654    if (set & clear) != 0 {
655        return ax_err!(
656            InvalidInput,
657            format_args!("can not set and clear the same bit in {:?}", control)
658        );
659    }
660    if (allowed1 & set) != set {
661        // failed if set 0-bits in allowed1
662        return ax_err!(
663            Unsupported,
664            format_args!("can not set bits {:#x} in {:?}", set, control)
665        );
666    }
667    if (allowed0 & clear) != 0 {
668        // failed if clear 1-bits in allowed0
669        return ax_err!(
670            Unsupported,
671            format_args!("can not clear bits {:#x} in {:?}", clear, control)
672        );
673    }
674    // SDM Vol. 3C, Section 31.5.1, Algorithm 3
675    let flexible = !allowed0 & allowed1; // therse bits can be either 0 or 1
676    let unknown = flexible & !(set | clear); // hypervisor untouched bits
677    let default = unknown & old_value; // these bits keep unchanged in old value
678    let fixed1 = allowed0; // these bits are fixed to 1
679    control.write(fixed1 | default | set)?;
680    Ok(())
681}

727pub fn inject_event(vector: u8, err_code: Option<u32>) -> AxResult {
728    // SDM Vol. 3C, Section 24.8.3
729    let err_code = if VmxInterruptionType::vector_has_error_code(vector) {
730        err_code.or_else(|| Some(VmcsReadOnly32::VMEXIT_INTERRUPTION_ERR_CODE.read().unwrap()))
731    } else {
732        None
733    };
734    let int_info = VmxInterruptInfo::from(vector, err_code);
735    if let Some(err_code) = int_info.err_code {
736        VmcsControl32::VMENTRY_EXCEPTION_ERR_CODE.write(err_code)?;
737    }
738    if int_info.int_type.is_soft() {
739        VmcsControl32::VMENTRY_INSTRUCTION_LEN
740            .write(VmcsReadOnly32::VMEXIT_INSTRUCTION_LEN.read()?)?;
741    }
742    VmcsControl32::VMENTRY_INTERRUPTION_INFO_FIELD.write(int_info.bits())?;
743    Ok(())
744}

758pub fn ept_violation_info() -> AxResult<NestedPageFaultInfo> {
759    // SDM Vol. 3C, Section 27.2.1, Table 27-7
760    let qualification = VmcsReadOnlyNW::EXIT_QUALIFICATION.read()?;
761    let fault_guest_paddr = VmcsReadOnly64::GUEST_PHYSICAL_ADDR.read()? as usize;
762    let mut access_flags = MappingFlags::empty();
763    if qualification.get_bit(0) {
764        access_flags |= MappingFlags::READ;
765    }
766    if qualification.get_bit(1) {
767        access_flags |= MappingFlags::WRITE;
768    }
769    if qualification.get_bit(2) {
770        access_flags |= MappingFlags::EXECUTE;
771    }
772    Ok(NestedPageFaultInfo {
773        access_flags,
774        fault_guest_paddr: GuestPhysAddr::from(fault_guest_paddr),
775    })
776}

778pub fn update_efer() -> AxResult {
779    use x86_64::registers::control::EferFlags;
780
781    let efer = VmcsGuest64::IA32_EFER.read()?;
782    let mut guest_efer = EferFlags::from_bits_truncate(efer);
783
784    if guest_efer.contains(EferFlags::LONG_MODE_ENABLE)
785        && guest_efer.contains(EferFlags::LONG_MODE_ACTIVE)
786    {
787        // debug!("Guest IA32_EFER LONG_MODE_ACTIVE is set, just return");
788        return Ok(());
789    }
790
791    guest_efer.set(EferFlags::LONG_MODE_ACTIVE, true);
792
793    // debug!(
794    //     "Guest IA32_EFER from {:?} update to {:?}",
795    //     EferFlags::from_bits_truncate(efer),
796    //     guest_efer
797    // );
798
799    VmcsGuest64::IA32_EFER.write(guest_efer.bits())?;
800
801    use controls::EntryControls as EntryCtrl;
802    set_control(
803        VmcsControl32::VMENTRY_CONTROLS,
804        Msr::IA32_VMX_TRUE_ENTRY_CTLS,
805        VmcsControl32::VMENTRY_CONTROLS.read()?,
806        (EntryCtrl::IA32E_MODE_GUEST).bits(),
807        0,
808    )?;
809
810    Ok(())
811}

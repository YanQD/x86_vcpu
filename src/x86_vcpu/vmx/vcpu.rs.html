60impl XState {
61    /// Create a new [`XState`] instance with current host state
62    fn new() -> Self {
63        // Check if XSAVE is available
64        let xsave_available = Self::xsave_available();
65        // Check if XSAVES and XRSTORS (as well as IA32_XSS) are available
66        let xsaves_available = if xsave_available {
67            Self::xsaves_available()
68        } else {
69            false
70        };
71
72        // Read XCR0 iff XSAVE is available
73        let xcr0 = if xsave_available {
74            unsafe { xcr0_read().bits() }
75        } else {
76            0
77        };
78        // Read IA32_XSS iff XSAVES is available
79        let xss = if xsaves_available {
80            Msr::IA32_XSS.read()
81        } else {
82            0
83        };
84
85        Self {
86            host_xcr0: xcr0,
87            guest_xcr0: xcr0,
88            host_xss: xss,
89            guest_xss: xss,
90            xsave_available,
91            xsaves_available,
92        }
93    }

120    /// Save the current host XCR0 and IA32_XSS values and load the guest values.
121    pub fn switch_to_guest(&mut self) {
122        unsafe {
123            if self.xsave_available {
124                self.host_xcr0 = xcr0_read().bits();
125                xcr0_write(Xcr0::from_bits_unchecked(self.guest_xcr0));
126
127                if self.xsaves_available {
128                    self.host_xss = Msr::IA32_XSS.read();
129                    Msr::IA32_XSS.write(self.guest_xss);
130                }
131            }
132        }
133    }

137        unsafe {
138            if self.xsave_available {
139                self.guest_xcr0 = xcr0_read().bits();
140                xcr0_write(Xcr0::from_bits_unchecked(self.host_xcr0));
141
142                if self.xsaves_available {
143                    self.guest_xss = Msr::IA32_XSS.read();
144                    Msr::IA32_XSS.write(self.host_xss);
145                }
146            }
147        }

261    /// Get CPU mode of the guest.
262    pub fn get_cpu_mode(&self) -> VmCpuMode {
263        let ia32_efer = Msr::IA32_EFER.read();
264        let cs_access_right = VmcsGuest32::CS_ACCESS_RIGHTS.read().unwrap();
265        let cr0 = VmcsGuestNW::CR0.read().unwrap();
266        if (ia32_efer & MSR_IA32_EFER_LMA_BIT) != 0 {
267            if (cs_access_right & 0x2000) != 0 {
268                // CS.L = 1
269                VmCpuMode::Mode64
270            } else {
271                VmCpuMode::Compatibility
272            }
273        } else if (cr0 & CR0_PE) != 0 {
274            VmCpuMode::Protected
275        } else {
276            VmCpuMode::Real
277        }

280    /// Run the guest. It returns when a vm-exit happens and returns the vm-exit if it cannot be handled by this [`VmxVcpu`] itself.
281    pub fn inner_run(&mut self) -> Option<VmxExitInfo> {
282        self.inject_pending_events().unwrap();
283
284        // Run guest
285        self.load_guest_xstate();
286
287        #[cfg(feature = "tracing")]
288        {
289            use crate::regs::GeneralRegistersDiff;
290            // Tracing, do a diff of the guest registers before entering the guest
291            let diff = GeneralRegistersDiff::new(self.guest_regs_exiting, self.guest_regs);
292            if !diff.is_same() {
293                debug!(
294                    "VCpu registers changed during handling VM-exit: {:#x?}",
295                    diff
296                );
297            } else {
298                debug!("VCpu registers unchanged during handling VM-exit");
299            }
300        }
301
302        unsafe {
303            if self.launched {
304                self.vmx_resume();
305            } else {
306                self.launched = true;
307                VmcsHostNW::RSP
308                    .write(&self.host_stack_top as *const _ as usize)
309                    .unwrap();
310
311                self.vmx_launch();
312            }
313        }
314        self.load_host_xstate();
315
316        #[cfg(feature = "tracing")]
317        {
318            self.guest_regs_exiting = self.guest_regs;
319        }
320
321        // Handle vm-exits
322        let exit_info = self.exit_info().unwrap();
323        // debug!("VM exit: {:#x?}", exit_info);
324
325        match self.builtin_vmexit_handler(&exit_info) {
326            Some(result) => {
327                if result.is_err() {
328                    panic!(
329                        "VmxVcpu failed to handle a VM-exit that should be handled by itself: {:?}, error {:?}, vcpu: {:#x?}",
330                        exit_info.exit_reason,
331                        result.unwrap_err(),
332                        self
333                    );
334                }
335
336                None
337            }
338            None => Some(exit_info),
339        }
340    }

407    /// Get Translate guest page table info
408    pub fn get_ptw_info(&self) -> GuestPageWalkInfo {
409        let top_entry = VmcsGuestNW::CR3.read().unwrap();
410        let level = self.get_paging_level();
411        let is_write_access = false;
412        let is_inst_fetch = false;
413        let is_user_mode_access = ((VmcsGuest32::SS_ACCESS_RIGHTS.read().unwrap() >> 5) & 0x3) == 3;
414        let mut pse = true;
415        let mut nxe =
416            (VmcsGuest64::IA32_EFER.read().unwrap() & EferFlags::NO_EXECUTE_ENABLE.bits()) != 0;
417        let wp = (VmcsGuestNW::CR0.read().unwrap() & Cr0Flags::WRITE_PROTECT.bits() as usize) != 0;
418        let is_smap_on = (VmcsGuestNW::CR4.read().unwrap()
419            & Cr4Flags::SUPERVISOR_MODE_ACCESS_PREVENTION.bits() as usize)
420            != 0;
421        let is_smep_on = (VmcsGuestNW::CR4.read().unwrap()
422            & Cr4Flags::SUPERVISOR_MODE_EXECUTION_PROTECTION.bits() as usize)
423            != 0;
424        let width: u32;
425        if level == 4 || level == 3 {
426            width = 9;
427        } else if level == 2 {
428            width = 10;
429            pse = VmcsGuestNW::CR4.read().unwrap() & Cr4Flags::PAGE_SIZE_EXTENSION.bits() as usize
430                != 0;
431            nxe = false;
432        } else {
433            width = 0;
434        }
435        GuestPageWalkInfo {
436            top_entry,
437            level,
438            width,
439            is_user_mode_access,
440            is_write_access,
441            is_inst_fetch,
442            pse,
443            wp,
444            nxe,
445            is_smap_on,
446            is_smep_on,
447        }
448    }

502    fn setup_io_bitmap(&mut self) -> AxResult {
503        // By default, I/O bitmap is set as `intercept_all`.
504        // Todo: these should be combined with emulated pio device management,
505        // in `modules/axvm/src/device/x86_64/mod.rs` somehow.
506        let io_to_be_intercepted = [
507            // // UART
508            // // 0x3f8..0x3f8 + 8, // COM1
509            // // We need to intercepted the access to COM2 ports.
510            // // Because we want to reserve this port for host Linux.
511            // 0x2f8..0x2f8 + 8, // COM2
512            // // 0x3e8..0x3e8 + 8, // COM3
513            // // 0x2e8..0x2e8 + 8, // COM4
514            // // Virual PIC
515            // 0x20..0x20 + 2, // PIC1
516            // 0xa0..0xa0 + 2, // PIC2
517            // // Debug Port
518            // // 0x80..0x80 + 1,   // Debug Port
519            // //
520            // 0x92..0x92 + 1, // system_control_a
521            // 0x61..0x61 + 1, // system_control_b
522            // // RTC
523            // 0x70..0x70 + 2, // CMOS
524            // 0x40..0x40 + 4, // PIT
525            // // 0xf0..0xf0 + 2,   // ports about fpu
526            // // 0x3d4..0x3d4 + 2, // ports about vga
527            // 0x87..0x87 + 1,   // port about dma
528            // 0x60..0x60 + 1,   // ports about ps/2 controller
529            // 0x64..0x64 + 1,   // ports about ps/2 controller
530            // 0xcf8..0xcf8 + 8, // PCI
531
532            // QEMU exit port
533            QEMU_EXIT_PORT..QEMU_EXIT_PORT + 1, // QEMU exit port
534        ];
535        for port_range in io_to_be_intercepted {
536            self.io_bitmap.set_intercept_of_range(
537                port_range.start as _,
538                port_range.count() as u32,
539                true,
540            );
541        }
542        Ok(())
543    }

581    fn setup_vmcs_host(&self) -> AxResult {
582        VmcsHost64::IA32_PAT.write(Msr::IA32_PAT.read())?;
583        VmcsHost64::IA32_EFER.write(Msr::IA32_EFER.read())?;
584
585        VmcsHostNW::CR0.write(Cr0::read_raw() as _)?;
586        VmcsHostNW::CR3.write(Cr3::read_raw().0.start_address().as_u64() as _)?;
587        VmcsHostNW::CR4.write(Cr4::read_raw() as _)?;
588
589        VmcsHost16::ES_SELECTOR.write(x86::segmentation::es().bits())?;
590        VmcsHost16::CS_SELECTOR.write(x86::segmentation::cs().bits())?;
591        VmcsHost16::SS_SELECTOR.write(x86::segmentation::ss().bits())?;
592        VmcsHost16::DS_SELECTOR.write(x86::segmentation::ds().bits())?;
593        VmcsHost16::FS_SELECTOR.write(x86::segmentation::fs().bits())?;
594        VmcsHost16::GS_SELECTOR.write(x86::segmentation::gs().bits())?;
595        VmcsHostNW::FS_BASE.write(Msr::IA32_FS_BASE.read() as _)?;
596        VmcsHostNW::GS_BASE.write(Msr::IA32_GS_BASE.read() as _)?;
597
598        let tr = unsafe { x86::task::tr() };
599        let mut gdtp = DescriptorTablePointer::<u64>::default();
600        let mut idtp = DescriptorTablePointer::<u64>::default();
601        unsafe {
602            dtables::sgdt(&mut gdtp);
603            dtables::sidt(&mut idtp);

674    fn setup_vmcs_control(&mut self, ept_root: HostPhysAddr, is_guest: bool) -> AxResult {
675        // Intercept NMI and external interrupts.
676        use super::vmcs::controls::*;
677        use PinbasedControls as PinCtrl;
678        let raw_cpuid = CpuId::new();
679
680        vmcs::set_control(
681            VmcsControl32::PINBASED_EXEC_CONTROLS,
682            Msr::IA32_VMX_TRUE_PINBASED_CTLS,
683            Msr::IA32_VMX_PINBASED_CTLS.read() as u32,
684            (PinCtrl::NMI_EXITING | PinCtrl::EXTERNAL_INTERRUPT_EXITING).bits(),
685            // (PinCtrl::NMI_EXITING | PinCtrl::VMX_PREEMPTION_TIMER).bits(),
686            // PinCtrl::NMI_EXITING.bits(),
687            0,
688        )?;
689
690        // Intercept all I/O instructions, use MSR bitmaps, activate secondary controls,
691        // disable CR3 load/store interception.
692        use PrimaryControls as CpuCtrl;
693        vmcs::set_control(
694            VmcsControl32::PRIMARY_PROCBASED_EXEC_CONTROLS,
695            Msr::IA32_VMX_TRUE_PROCBASED_CTLS,
696            Msr::IA32_VMX_PROCBASED_CTLS.read() as u32,
697            (CpuCtrl::USE_IO_BITMAPS | CpuCtrl::USE_MSR_BITMAPS | CpuCtrl::SECONDARY_CONTROLS)

701                | CpuCtrl::CR8_LOAD_EXITING
702                | CpuCtrl::CR8_STORE_EXITING)
703                .bits(),
704        )?;
705
706        // Enable EPT, RDTSCP, INVPCID, and unrestricted guest.
707        use SecondaryControls as CpuCtrl2;
708        let mut val =
709            // CpuCtrl2::VIRTUALIZE_APIC | 
710            CpuCtrl2::ENABLE_EPT | CpuCtrl2::UNRESTRICTED_GUEST;
711        if let Some(features) = raw_cpuid.get_extended_processor_and_feature_identifiers() {
712            if features.has_rdtscp() {
713                val |= CpuCtrl2::ENABLE_RDTSCP;
714            }
715        }
716        if let Some(features) = raw_cpuid.get_extended_feature_info() {
717            if features.has_invpcid() {
718                val |= CpuCtrl2::ENABLE_INVPCID;
719            }
720        }
721        if let Some(features) = raw_cpuid.get_extended_state_info() {
722            if features.has_xsaves_xrstors() {
723                val |= CpuCtrl2::ENABLE_XSAVES_XRSTORS;
724            }
725        }
726        vmcs::set_control(
727            VmcsControl32::SECONDARY_PROCBASED_EXEC_CONTROLS,
728            Msr::IA32_VMX_PROCBASED_CTLS2,
729            Msr::IA32_VMX_PROCBASED_CTLS2.read() as u32,
730            val.bits(),
731            0,
732        )?;
733
734        // Switch to 64-bit host, acknowledge interrupt info, switch IA32_PAT/IA32_EFER on VM exit.
735        use ExitControls as ExitCtrl;
736        vmcs::set_control(
737            VmcsControl32::VMEXIT_CONTROLS,
738            Msr::IA32_VMX_TRUE_EXIT_CTLS,
739            Msr::IA32_VMX_EXIT_CTLS.read() as u32,
740            (ExitCtrl::HOST_ADDRESS_SPACE_SIZE
741                | ExitCtrl::ACK_INTERRUPT_ON_EXIT
742                | ExitCtrl::SAVE_IA32_PAT
743                | ExitCtrl::LOAD_IA32_PAT
744                | ExitCtrl::SAVE_IA32_EFER
745                | ExitCtrl::LOAD_IA32_EFER)
746                .bits(),
747            0,
748        )?;
749
750        let mut val = EntryCtrl::LOAD_IA32_PAT | EntryCtrl::LOAD_IA32_EFER;
751
752        if !is_guest {
753            // IA-32e mode guest
754            // On processors that support Intel 64 architecture, this control determines whether the logical processor is in IA-32e mode after VM entry.
755            // Its value is loaded into IA32_EFER.LMA as part of VM entry.
756            val |= EntryCtrl::IA32E_MODE_GUEST;
757        }
758
759        // Load guest IA32_PAT/IA32_EFER on VM entry.
760        use EntryControls as EntryCtrl;
761        vmcs::set_control(
762            VmcsControl32::VMENTRY_CONTROLS,
763            Msr::IA32_VMX_TRUE_ENTRY_CTLS,
764            Msr::IA32_VMX_ENTRY_CTLS.read() as u32,
765            val.bits(),
766            0,
767        )?;
768
769        vmcs::set_ept_pointer(ept_root)?;
770
771        // No MSR switches if hypervisor doesn't use and there is only one vCPU.
772        VmcsControl32::VMEXIT_MSR_STORE_COUNT.write(0)?;
773        VmcsControl32::VMEXIT_MSR_LOAD_COUNT.write(0)?;
774        VmcsControl32::VMENTRY_MSR_LOAD_COUNT.write(0)?;
775
776        // VmcsControlNW::CR4_GUEST_HOST_MASK.write(0)?;
777        VmcsControl32::CR3_TARGET_COUNT.write(0)?;
778
779        // Pass-through exceptions (except #UD(6)), don't use I/O bitmap, set MSR bitmaps.
780        let exception_bitmap: u32 = 1 << 6;
781
782        self.setup_io_bitmap()?;
783
784        VmcsControl32::EXCEPTION_BITMAP.write(exception_bitmap)?;
785        VmcsControl64::IO_BITMAP_A_ADDR.write(self.io_bitmap.phys_addr().0.as_usize() as _)?;
786        VmcsControl64::IO_BITMAP_B_ADDR.write(self.io_bitmap.phys_addr().1.as_usize() as _)?;
787        VmcsControl64::MSR_BITMAPS_ADDR.write(self.msr_bitmap.phys_addr().as_usize() as _)?;
788
789        // VmcsControl64::APIC_ACCESS_ADDR.write(
790        //     EmulatedLocalApic::<H::MmHal, DummyHal>::virtual_apic_access_addr().as_usize() as _,
791        // )?;
792        Ok(())
793    }

795    fn get_paging_level(&self) -> usize {
796        let mut level: u32 = 0; // non-paging
797        let cr0 = VmcsGuestNW::CR0.read().unwrap();
798        let cr4 = VmcsGuestNW::CR4.read().unwrap();
799        let efer = VmcsGuest64::IA32_EFER.read().unwrap();
800        // paging is enabled
801        if cr0 & Cr0Flags::PAGING.bits() as usize != 0 {
802            if cr4 & Cr4Flags::PHYSICAL_ADDRESS_EXTENSION.bits() as usize != 0 {
803                // is long mode
804                if efer & EferFlags::LONG_MODE_ACTIVE.bits() != 0 {
805                    level = 4;
806                } else {
807                    level = 3;
808                }
809            } else {
810                level = 2;
811            }
812        }
813        level as usize
814    }

820    fn set_cr(&mut self, cr_idx: usize, val: u64) {
821        (|| -> AxResult {
822            // debug!("set guest CR{} to val {:#x}", cr_idx, val);
823            match cr_idx {
824                0 => {
825                    // Retrieve/validate restrictions on CR0
826                    //
827                    // In addition to what the VMX MSRs tell us, make sure that
828                    // - NW and CD are kept off as they are not updated on VM exit and we
829                    //   don't want them enabled for performance reasons while in root mode
830                    // - PE and PG can be freely chosen (by the guest) because we demand
831                    //   unrestricted guest mode support anyway
832                    // - ET is ignored
833                    let must0 = Msr::IA32_VMX_CR0_FIXED1.read()
834                        & !(Cr0Flags::NOT_WRITE_THROUGH | Cr0Flags::CACHE_DISABLE).bits();
835                    let must1 = Msr::IA32_VMX_CR0_FIXED0.read()
836                        & !(Cr0Flags::PAGING | Cr0Flags::PROTECTED_MODE_ENABLE).bits();
837                    VmcsGuestNW::CR0.write(((val & must0) | must1) as _)?;
838                    VmcsControlNW::CR0_READ_SHADOW.write(val as _)?;
839                    VmcsControlNW::CR0_GUEST_HOST_MASK.write((must1 | !must0) as _)?;
840                }
841                3 => VmcsGuestNW::CR3.write(val as _)?,
842                4 => {
843                    // Retrieve/validate restrictions on CR4
844                    let must0 = Msr::IA32_VMX_CR4_FIXED1.read();
845                    let must1 = Msr::IA32_VMX_CR4_FIXED0.read();
846                    let val = val | Cr4Flags::VIRTUAL_MACHINE_EXTENSIONS.bits();
847                    VmcsGuestNW::CR4.write(((val & must0) | must1) as _)?;
848                    VmcsControlNW::CR4_READ_SHADOW.write(val as _)?;
849                    VmcsControlNW::CR4_GUEST_HOST_MASK.write((must1 | !must0) as _)?;
850                }
851                _ => unreachable!(),
852            };
853            Ok(())
854        })()
855        .expect("Failed to write guest control register")
856    }
857
858    #[allow(dead_code)]
859    fn cr(&self, cr_idx: usize) -> usize {
860        (|| -> AxResult<usize> {
861            Ok(match cr_idx {
862                0 => VmcsGuestNW::CR0.read()?,
863                3 => VmcsGuestNW::CR3.read()?,
864                4 => {
865                    let host_mask = VmcsControlNW::CR4_GUEST_HOST_MASK.read()?;
866                    (VmcsControlNW::CR4_READ_SHADOW.read()? & host_mask)
867                        | (VmcsGuestNW::CR4.read()? & !host_mask)
868                }
869                _ => unreachable!(),
870            })
871        })()
872        .expect("Failed to read guest control register")
873    }

877macro_rules! vmx_entry_with {
878    ($instr:literal) => {
879        naked_asm!(
880            save_regs_to_stack!(),                  // save host status
881            "mov    [rdi + {host_stack_size}], rsp", // save current RSP to Vcpu::host_stack_top
882            "mov    rsp, rdi",                      // set RSP to guest regs area

885            "jmp    {failed}",
886            host_stack_size = const size_of::<GeneralRegisters>(),
887            failed = sym Self::vmx_entry_failed,
888            // options(noreturn),
889        )
890    }

962    /// Handle vm-exits than can and should be handled by [`VmxVcpu`] itself.
963    ///
964    /// Return the result or None if the vm-exit was not handled.
965    fn builtin_vmexit_handler(&mut self, exit_info: &VmxExitInfo) -> Option<AxResult> {
966        const X2APIC_MSR_BASE: u32 = 0x800;
967        const X2APIC_MSR_END: u32 = 0x8ff; // SDM says 0x8ff, but actually 0x83f, we respect the SDM here.
968        // Following vm-exits are handled here:
969        // - interrupt window: turn off interrupt window;
970        // - xsetbv: set guest xcr;
971        // - cr access: just panic;
972        match exit_info.exit_reason {
973            VmxExitReason::INTERRUPT_WINDOW => Some(self.set_interrupt_window(false)),
974            VmxExitReason::PREEMPTION_TIMER => Some(self.handle_vmx_preemption_timer()),
975            VmxExitReason::XSETBV => Some(self.handle_xsetbv()),
976            VmxExitReason::CR_ACCESS => Some(self.handle_cr()),
977            VmxExitReason::CPUID => Some(self.handle_cpuid()),
978            msr_rw @ (VmxExitReason::MSR_READ | VmxExitReason::MSR_WRITE)
979                if {
980                    let msr = self.regs().rcx as u32;
981                    msr >= X2APIC_MSR_BASE && msr <= X2APIC_MSR_END
982                } =>
983            {
984                Some(self.handle_apic_msr_access(
985                    msr_rw == VmxExitReason::MSR_WRITE,
986                    self.regs().rcx as u32,

988            }
989            VmxExitReason::APIC_ACCESS => Some(self.handle_apic_access(exit_info)),
990            _ => None,
991        }

1071    fn handle_cr(&mut self) -> AxResult {
1072        const VM_EXIT_INSTR_LEN_MV_TO_CR: u8 = 3;
1073
1074        let cr_access_info = vmcs::cr_access_info()?;
1075
1076        let reg = cr_access_info.gpr;
1077        let cr = cr_access_info.cr_number;
1078
1079        match cr_access_info.access_type {
1080            /* move to cr */
1081            0 => {
1082                let val = if reg == 4 {
1083                    self.stack_pointer() as u64
1084                } else {
1085                    self.guest_regs.get_reg_of_index(reg)
1086                };
1087                if cr == 0 || cr == 4 {
1088                    self.advance_rip(VM_EXIT_INSTR_LEN_MV_TO_CR)?;
1089                    /* TODO: check for #GP reasons */
1090                    self.set_cr(cr as usize, val);
1091
1092                    if cr == 0 && Cr0Flags::from_bits_truncate(val).contains(Cr0Flags::PAGING) {
1093                        vmcs::update_efer()?;
1094                    }
1095                    return Ok(());
1096                }
1097            }
1098            _ => {}
1099        };
1100
1101        panic!(
1102            "Guest's access to cr not allowed: {:#x?}, {:#x?}",
1103            self, cr_access_info
1104        );
1105    }

1107    fn handle_cpuid(&mut self) -> AxResult {
1108        use raw_cpuid::{CpuIdResult, cpuid};
1109
1110        const VM_EXIT_INSTR_LEN_CPUID: u8 = 2;
1111        const LEAF_FEATURE_INFO: u32 = 0x1;
1112        const LEAF_STRUCTURED_EXTENDED_FEATURE_FLAGS_ENUMERATION: u32 = 0x7;
1113        const LEAF_PROCESSOR_EXTENDED_STATE_ENUMERATION: u32 = 0xd;
1114        const EAX_FREQUENCY_INFO: u32 = 0x16;
1115        const LEAF_HYPERVISOR_INFO: u32 = 0x4000_0000;
1116        const LEAF_HYPERVISOR_FEATURE: u32 = 0x4000_0001;
1117        const VENDOR_STR: &[u8; 12] = b"RVMRVMRVMRVM";
1118        let vendor_regs = unsafe { &*(VENDOR_STR.as_ptr() as *const [u32; 3]) };
1119
1120        let regs_clone = self.regs_mut().clone();
1121        let function = regs_clone.rax as u32;
1122        let res = match function {
1123            LEAF_FEATURE_INFO => {
1124                const FEATURE_VMX: u32 = 1 << 5;
1125                const FEATURE_HYPERVISOR: u32 = 1 << 31;
1126                const FEATURE_MCE: u32 = 1 << 7;
1127                let mut res = cpuid!(regs_clone.rax, regs_clone.rcx);
1128                res.ecx &= !FEATURE_VMX;
1129                res.ecx |= FEATURE_HYPERVISOR;
1130                res.eax &= !FEATURE_MCE;
1131                res
1132            }
1133            // See SDM Table 3-8. Information Returned by CPUID Instruction (Contd.)
1134            LEAF_STRUCTURED_EXTENDED_FEATURE_FLAGS_ENUMERATION => {
1135                let mut res = cpuid!(regs_clone.rax, regs_clone.rcx);
1136                if regs_clone.rcx == 0 {
1137                    // Bit 05: WAITPKG.
1138                    res.ecx.set_bit(5, false); // clear waitpkg
1139                    // Bit 16: LA57. Supports 57-bit linear addresses and five-level paging if 1.
1140                    res.ecx.set_bit(16, false); // clear LA57
1141                }
1142
1143                res
1144            }
1145            LEAF_PROCESSOR_EXTENDED_STATE_ENUMERATION => {
1146                self.load_guest_xstate();
1147                let res = cpuid!(regs_clone.rax, regs_clone.rcx);
1148                self.load_host_xstate();
1149
1150                res
1151            }
1152            LEAF_HYPERVISOR_INFO => CpuIdResult {
1153                eax: LEAF_HYPERVISOR_FEATURE,
1154                ebx: vendor_regs[0],
1155                ecx: vendor_regs[1],
1156                edx: vendor_regs[2],
1157            },
1158            LEAF_HYPERVISOR_FEATURE => CpuIdResult {
1159                eax: 0,
1160                ebx: 0,
1161                ecx: 0,
1162                edx: 0,
1163            },
1164            EAX_FREQUENCY_INFO => {
1165                /// Timer interrupt frequencyin Hz.
1166                /// Todo: this should be the same as `axconfig::TIMER_FREQUENCY` defined in ArceOS's config file.
1167                const TIMER_FREQUENCY_MHZ: u32 = 3_000;
1168                let mut res = cpuid!(regs_clone.rax, regs_clone.rcx);
1169                if res.eax == 0 {
1170                    warn!(
1171                        "handle_cpuid: Failed to get TSC frequency by CPUID, default to {} MHz",
1172                        TIMER_FREQUENCY_MHZ
1173                    );
1174                    res.eax = TIMER_FREQUENCY_MHZ;
1175                }
1176                res
1177            }
1178            _ => cpuid!(regs_clone.rax, regs_clone.rcx),
1179        };
1180
1181        trace!(
1182            "VM exit: CPUID({:#x}, {:#x}): {:?}",
1183            regs_clone.rax, regs_clone.rcx, res
1184        );
1185
1186        let regs = self.regs_mut();
1187        regs.rax = res.eax as _;
1188        regs.rbx = res.ebx as _;
1189        regs.rcx = res.ecx as _;
1190        regs.rdx = res.edx as _;
1191        self.advance_rip(VM_EXIT_INSTR_LEN_CPUID)?;
1192
1193        Ok(())
1194    }

1196    fn handle_xsetbv(&mut self) -> AxResult {
1197        const XCR_XCR0: u64 = 0;
1198        const VM_EXIT_INSTR_LEN_XSETBV: u8 = 3;
1199
1200        let index = self.guest_regs.rcx.get_bits(0..32);
1201        let value = self.guest_regs.rdx.get_bits(0..32) << 32 | self.guest_regs.rax.get_bits(0..32);
1202
1203        // TODO: get host-supported xcr0 mask by cpuid and reject any guest-xsetbv violating that
1204        if index == XCR_XCR0 {
1205            Xcr0::from_bits(value)
1206                .and_then(|x| {
1207                    if !x.contains(Xcr0::XCR0_FPU_MMX_STATE) {
1208                        return None;
1209                    }
1210
1211                    if x.contains(Xcr0::XCR0_AVX_STATE) && !x.contains(Xcr0::XCR0_SSE_STATE) {
1212                        return None;
1213                    }
1214
1215                    if x.contains(Xcr0::XCR0_BNDCSR_STATE) ^ x.contains(Xcr0::XCR0_BNDREG_STATE) {
1216                        return None;
1217                    }
1218
1219                    if x.contains(Xcr0::XCR0_OPMASK_STATE)
1220                        || x.contains(Xcr0::XCR0_ZMM_HI256_STATE)
1221                        || x.contains(Xcr0::XCR0_HI16_ZMM_STATE)
1222                        || !x.contains(Xcr0::XCR0_AVX_STATE)
1223                        || !x.contains(Xcr0::XCR0_OPMASK_STATE)
1224                        || !x.contains(Xcr0::XCR0_ZMM_HI256_STATE)
1225                        || !x.contains(Xcr0::XCR0_HI16_ZMM_STATE)
1226                    {
1227                        return None;
1228                    }
1229
1230                    Some(x)
1231                })
1232                .ok_or(ax_err_type!(InvalidInput))
1233                .and_then(|x| {
1234                    self.xstate.guest_xcr0 = x.bits();
1235                    self.advance_rip(VM_EXIT_INSTR_LEN_XSETBV)
1236                })
1237        } else {
1238            // xcr0 only
1239            ax_err!(Unsupported, "only xcr0 is supported")
1240        }
1241    }

1259fn get_tr_base(tr: SegmentSelector, gdt: &DescriptorTablePointer<u64>) -> u64 {
1260    let index = tr.index() as usize;
1261    let table_len = (gdt.limit as usize + 1) / core::mem::size_of::<u64>();
1262    let table = unsafe { core::slice::from_raw_parts(gdt.base, table_len) };
1263    let entry = table[index];
1264    if entry & (1 << 47) != 0 {
1265        // present
1266        let base_low = entry.get_bits(16..40) | entry.get_bits(56..64) << 24;
1267        let base_high = table[index + 1] & 0xffff_ffff;
1268        base_low | base_high << 32
1269    } else {
1270        // no present
1271        0
1272    }

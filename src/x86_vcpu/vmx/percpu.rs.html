12/// Represents the per-CPU state for Virtual Machine Extensions (VMX).
13///
14/// This structure holds the state information specific to a CPU core
15/// when operating in VMX mode, including the VMCS revision identifier and
16/// the VMX region.
17pub struct VmxPerCpuState<H: AxVCpuHal> {
18    /// The VMCS (Virtual Machine Control Structure) revision identifier.
19    ///
20    /// This identifier is used to ensure compatibility between the software
21    /// and the specific version of the VMCS that the CPU supports.
22    pub(crate) vmcs_revision_id: u32,
23
24    /// The VMX region for this CPU.
25    ///
26    /// This region typically contains the VMCS and other state information
27    /// required for managing virtual machines on this particular CPU.
28    vmx_region: VmxRegion<H::MmHal>,
29}
30
31impl<H: AxVCpuHal> AxArchPerCpu for VmxPerCpuState<H> {
32    fn new(_cpu_id: usize) -> AxResult<Self> {
33        Ok(Self {
34            vmcs_revision_id: 0,
35            vmx_region: unsafe { VmxRegion::uninit() },
36        })
37    }
38
39    fn is_enabled(&self) -> bool {
40        Cr4::read().contains(Cr4Flags::VIRTUAL_MACHINE_EXTENSIONS)
41    }
42
43    fn hardware_enable(&mut self) -> AxResult {
44        if !has_hardware_support() {
45            return ax_err!(Unsupported, "CPU does not support feature VMX");
46        }
47        if self.is_enabled() {
48            return ax_err!(ResourceBusy, "VMX is already turned on");
49        }
50
51        // Enable XSAVE/XRSTOR.
52        super::vcpu::XState::enable_xsave();
53
54        // Enable VMXON, if required.
55        let ctrl = FeatureControl::read();
56        let locked = ctrl.contains(FeatureControlFlags::LOCKED);
57        let vmxon_outside = ctrl.contains(FeatureControlFlags::VMXON_ENABLED_OUTSIDE_SMX);
58        if !locked {
59            FeatureControl::write(
60                ctrl | FeatureControlFlags::LOCKED | FeatureControlFlags::VMXON_ENABLED_OUTSIDE_SMX,
61            )
62        } else if !vmxon_outside {
63            return ax_err!(Unsupported, "VMX disabled by BIOS");
64        }
65
66        // Check control registers are in a VMX-friendly state. (SDM Vol. 3C, Appendix A.7, A.8)
67        {
68            use Msr::*;
69            let cr0_value = Cr0::read().bits();
70            let cr0_fixed0 = IA32_VMX_CR0_FIXED0.read();
71            let cr0_fixed1 = IA32_VMX_CR0_FIXED1.read();
72            if !((!cr0_fixed0 | cr0_value) != 0 && (cr0_fixed1 | !cr0_value) != 0) {
73                return ax_err!(BadState, "host CR0 is not valid in VMX operation");
74            }
75
76            let cr4_value = Cr4::read().bits();
77            let cr4_fixed0 = IA32_VMX_CR4_FIXED0.read();
78            let cr4_fixed1 = IA32_VMX_CR4_FIXED1.read();
79            if !((!cr4_fixed0 | cr4_value) != 0 && (cr4_fixed1 | !cr4_value) != 0) {
80                return ax_err!(BadState, "host CR4 is not valid in VMX operation");
81            }
82        }
83
84        // Get VMCS revision identifier in IA32_VMX_BASIC MSR.
85        let vmx_basic = VmxBasic::read();
86        if vmx_basic.region_size as usize != PAGE_SIZE {
87            return ax_err!(Unsupported);
88        }
89        if vmx_basic.mem_type != VmxBasic::VMX_MEMORY_TYPE_WRITE_BACK {
90            return ax_err!(Unsupported);
91        }
92        if vmx_basic.is_32bit_address {
93            return ax_err!(Unsupported);
94        }
95        if !vmx_basic.io_exit_info {
96            return ax_err!(Unsupported);
97        }
98        if !vmx_basic.vmx_flex_controls {
99            return ax_err!(Unsupported);
100        }
101        self.vmcs_revision_id = vmx_basic.revision_id;
102        self.vmx_region = VmxRegion::new(self.vmcs_revision_id, false)?;
103
104        unsafe {
105            // Enable VMX using the VMXE bit.
106            Cr4::write(Cr4::read() | Cr4Flags::VIRTUAL_MACHINE_EXTENSIONS);
107            // Execute VMXON.
108            vmx::vmxon(self.vmx_region.phys_addr().as_usize() as _).map_err(|err| {
109                ax_err_type!(
110                    BadState,
111                    format_args!("VMX instruction vmxon failed: {:?}", err)
112                )
113            })?;
114        }
115        info!("[AxVM] succeeded to turn on VMX.");
116
117        Ok(())
118    }
119
120    fn hardware_disable(&mut self) -> AxResult {
121        if !self.is_enabled() {
122            return ax_err!(BadState, "VMX is not enabled");
123        }
124
125        unsafe {
126            // Execute VMXOFF.
127            vmx::vmxoff().map_err(|err| {
128                ax_err_type!(
129                    BadState,
130                    format_args!("VMX instruction vmxoff failed: {:?}", err)
131                )
132            })?;
133            // Remove VMXE bit in CR4.
134            Cr4::update(|cr4| cr4.remove(Cr4Flags::VIRTUAL_MACHINE_EXTENSIONS));
135        };
136        info!("[AxVM] succeeded to turn off VMX.");
137
138        self.vmx_region = unsafe { VmxRegion::uninit() };
139        Ok(())
140    }
141}
